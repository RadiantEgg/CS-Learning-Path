char *my_fgets(char *str, int n, FILE *stream)
{
    int c;
    char *cs = str;
    
    while (--n > 0 && (c = fgetc(stream)) != NULL) {
        if ((*cs++ = c) == '\n'))
            break;
    }
    *cs = '\0';

    if (cs == str && c == EOF)
        return NULL;

    return str;
}


int my_puts(const char *str, FILE *stream)
{
    while (*str != '\0') {
        if (fputc(*str, stream) == EOF)
            return EOF;
        
        str++;
    }
    return 0;
}

- 数组序列化: 先写入数组长度，再写入数据数据块
```c
void serialize_array(const char *filename, void *array, int size, int count)
{
    FILE *fp = fopen(filename, "wb");
    if (!fp) {
        perror("写入失败");
        return;
    }

    // 先写入数组长度
    fwrite(&count, sizeof(int), 1, fp);

    // 再写入整个数组数据块
    fwrite(array, size, count, fp);
    fclose(fp);
}

// 反序列化 
void *deserialize_array(const char *filename, int size, int *count)
{
    FILE *fp = fopen(filename, "rb");
    if (!fp) {
        ferror("读取失败");
        return;
    }

    fread(count, sizeof(int), 1, fp);

    void *array = malloc(*count * size);
    if (!array) {
        fclose(fp);
        return NULL;
    }

    fread(array, size, *count, fp);
    fclose(fp);
    return array;
}

- 结构体对齐：在结构体成员之间和结构体末尾插入padding字节

void serialize_student(const char *filename,const Student *st)
{
    FILE *fp = fopen(filename, "wb");
    if (fp) {
        fwrite(&st->id, sizeof(st->id), 1, fp);

        int len = strlen(st->name);
        fwrite(&len, sizeof(len), 1, fp);

        fwrite(st->name, sizeof(char), len, fp);
        fclose(fp);
    }
}

Student *deserialize_student(const char *filename)
{
    FILE * fp = fopen(filename, "rb");

    Student *st = malloc(sizeof(Student));

    size_t len;

    fread(&st->id, sizeof(st->id), 1, fp);

    fread(&len, sizeof(len), 1, fp);
    
    st->name = malloc(len + 1);
    fread(st->name, 1, len, fp);
    st->name[len] = '\0';

    fclose(fp);

    return st;
}
