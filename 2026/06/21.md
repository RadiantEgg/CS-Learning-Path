- 格式化读写和字符串读写
```c
// fgets -> fscanf
fscanf(fp, "%d %s", &id, name);

if (fgets(buf, sizeof(buf))) {
    sscanf(buf, "%d %s", &id, name);
}

// fputs -> fprintf
fprintf(fp, "%d %s", id, name);

sprintf(buf, "%d %s", id, name);
fputs(buf, fp);

```

- 模拟读取字符串
```c
char *my_fgets(char *str, int n, FILE *stream)
{
    int c;
    char *cs = str;
    
    while (--n > 0 && (c = fgetc(stream)) != NULL) {
        if ((*cs++ = c) == '\n'))
            break;
    }
    *cs = '\0';

    if (cs == str && c == EOF)
        return NULL;

    return str;
}


int my_puts(const char *str, FILE *stream)
{
    while (*str != '\0') {
        if (fputc(*str, stream) == EOF)
            return EOF;
        
        str++;
    }
    return 0;
}


```
