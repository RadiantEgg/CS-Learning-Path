- 通用数据交换函数
```c
#include <stdio.h>
#include <stdlib.h>

void swap(void *a, void *b, size_t size)     // 两块大小为size的内存块
{
    char *pa = a;   //逐字节复制
    char *pb = b;

    while (size--) {
        char tmp = *pa;
        *pa = *pb;
        *pb = tmp;

        pa++;
        pb++;
    }
}
```

```c
void qsort(int *a, int left, int right)
{
    if (left >= right)
        return;

    int last = left;
    int mid = (left + right) / 2;
    
    swap(a, left, mid);

    for (int i = left + 1; i <= right; i++) {
        if (a[i] < a[left])
            swap(a, ++last, i);
    }

    swap(a, left, last);

    qsort(a, left, last - 1);
    qsort(a, last + 1, right);
}
```
