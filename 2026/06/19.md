```c
#include <stdio.h>
#include <stdlib.h>

void swap(void *a, void *b, size_t size)     // 两块大小为size的内存块
{
    char *pa = a;   //逐字节复制
    char *pb = b;

    while (size--) {
        char tmp = *pa;
        *pa = *pb;
        *pb = tmp;

        pa++;
        pb++;
    }
}
```

```c
void qsort(int *a, int left, int right)
{
    if (left >= right)
        return;

    int last = left;
    int mid = (left + right) / 2;
    
    swap(a, left, mid);

    for (int i = left + 1; i <= right; i++) {
        if (a[i] < a[left])
            swap(a, ++last, i);
    }

    swap(a, left, last);

    qsort(a, left, last - 1);
    qsort(a, last + 1, right);
}
```

- VA_LIST 宏定义 （可变参数）
    - va_list: 参数游标（指向参数的指针）
    - va_start: 指向第一个可变参数
    - va_arg: 读取当前参数并移动游标
    - va_end: 结束参数获取
```c

#include <stdarg.h>

int max_int(int n, ...) //  ... 表示参数可变
{
    va_list ap;
    va_start(ap, n);
    max = va_arg(ap, int);
    for (int i = 1; i <n; i++) {
        int cur = va_arg(ap, int);
        max = (cur > max) ? cur : max;
    }
    va_end(ap);
    return max;
}
```

```c
#include <stdio.h>
#include <stdint.h>

// 内存池的定义
#define POOL_SIZE 10    // 内存池中内存块的数量
#define BLOCK_SIZE 32   // 每个内存块的大小

typedef struct {
    unit8_t data[POOL_SIZE][BLOCK_SIZE];  // 实际存储数据的二维数组
    int next_free[POOL_SIZE];    // 存储下一个空闲块索引的数组
    int first_free;     // 当前第一个可用的空闲块
    int free_count;     // 剩余空闲块数量
} StaticPool;

// 内存池初始化
void pool_init(StaticPool *pool)
{
    for (int i = 0; i < POOL_SIZE - 1; i++) 
        pool->next_free[i] = i + 1;     // 每个块指向下一个

    pool->next_free[POOL_SIZE - 1] = -1  // 最后一个块为结尾

    pool->first = 0;
    pool->free_count = POOL_SIZE;
}

// 分配内存
void *pool_alloc(StaticPool *pool)
{
    if (pool->first_free == -1) {
        printf("Error: Pool out of memory\n");
        return NULL;
    }

    // 获取当前空闲块指针
    void *res = &(pool->data[pool->first_free]);
    // 更新空闲指针
    pool->first_free = pool->next_free[pool->first_free];
    pool->free_count--;
    return res;
}

// 释放内存
void pool_free(StaticPool *pool, void *prt)
{

        return;
    }

    // 释放的内存块插回空闲表头部
    pool->next_free[index] = pool->first_free;
    pool->first_free = index;
    pool->free_count--;
}
```
