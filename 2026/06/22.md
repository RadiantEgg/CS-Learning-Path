```c
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;


// 链表的深拷贝
void serialize_link(const char *filename, const Node *head)
{
    FILE *fp = fopen(filename, "wb");

    int count = 0;
    Node *cur = head;
    while (cur) {
        count++;
        cur = cur->next;
    }

    fwrite(&count, sizeof(count), 1, fp);

    cur = head;
    for (int i = 0; i < count; i++) {
        fwrite(&cur->data, sizeof(int), 1, fp);
        cur = cur->next;
    }

    fclose(fp);
}

// 深度反序列化
void *deserialize_link(const char *filename)
{
    FILE *fp = fopen(filename, "rb");

    int count;
    fread(&count, sizeof(int), 1, fp);

    Node *head = NULL;
    Node *tail = NULL;
    head->next = tail;
    for (int i = 0; i < count; i++) {
        int data;
        fread(&data, sizeof(int), 1, fp);

        Node *newnode = malloc(sizeof(Node));
        newnode->data = data;
        newnode->next = NULL;

        if (head == NULL) {
            head = newnode;
            tail = newnode;
        } else {
            tail->next = newnode;
            tail = newnode;
        }
        fclose(fp);
    }
}

// 魔数是位于文件开头的几个固定字节用于显示文件格式

// 判断图片是否为PNG格式
// PNG文件的魔数：8字节，十六进制为：89 50 4E 47 0D 0A 1A 0A

/*
    不同平台int字节数不一样
    引入<stdint.h>
    定义一组精确宽度宽度的类型，类型会被定义为在使用平台上恰好满足要求的类型

*/ 


int judge_png(const char *filename)
{
    static const uint8_t png_magic[8] = {0x89, 0x50, 0x4E, 0x47, 0x0D,0x0A, 0x1A, 0x0A};
    FILE *fp = fopen(filename, "rb");

    uint8_t header[8];
    size_t bytes_read = fread(header, 1, 8, fp);
    fclose(fp);

    return  memcmp(header, png_magic, 8) == 0;
}

#define MY_MAGIC_NUMBER 0x55535443  // USTC 的十六进制
#define CURRENT_VRESION 1

typedef struct {
    int id;
    char name[20];
    float score;
} Student;


typedef struct {
    uint32_t magic; // 魔数
    uint16_t version; // 版本号
} FileHeader;

void save_data(const char *filename, const Student *ps)
{
    FILE *fp = fopen(filename, "wb");

    FileHeader header = {MY_MAGIC_NUMBER, CURRENT_VRESION};

    // 写入文件头，包括魔数和版本号
    fwrite(&header, sizeof(FileHeader), 1, fp);

    fclose(fp);
}

void load_data(const char *filename, Student *ps)
{
    FILE *fp = fopen(filename, "rb");

    FileHeader header;

    if (fread(&header, sizeof(FileHeader), 1, fp) != 1) {
        fclose(fp);
        return -2;  // 错误：无法读取文件头
    }
    if (header.magic != MY_MAGIC_NUMBER) {
        fclose(fp);
        return -3;  // 错误：不是ustc文件
    }
    if (header.version > CURRENT_VRESION) {
        fclose(fp);
        return -4;  // 错误：存档版本过高，需要更新程序
    }

    fclose(fp);
    return 0;
}
```

- 文件读写位置指针
    - 位置指针定位到指定位置，随机读写`int fseek(FILE *stream, long offset, int origin);`
        - origin 文件位置指针移动参考点
            - 文件头：SEEK_SET 0
            - 当前位置：SEEK_CRU 1
            - 文件尾：SEEK_END 2
        - offset 从参考点的位移量，可正（向文件末尾移动），可负（向文件开头移动）或0（不动）
        ```c
        // 逆向输出文件所有字节
        for (count = 1; count <= size; count++) {
            fseek(fp, -count, SEEK_END);
            ch = fgetc(fp);
        }

        ```
    - 位置指针重新定位到开头 `void rewind(FILE *stream)`
    等价于`fseek(stream, 0, SEEK_SET)`
    - 在读写切换的时候需要fseek 到当前位置来刷新操作状态：写->读或读->写

- 获取文件位置指针当前位置：`long ftell(FILE *stream)`
    - 成功返回相对文件开头的位置，失败返回-1L
```c
// ftell 与 fseek 配合使用可以在操作完之后返回原来位置
long file_pos = ftell(fp);
fseek(fp, file_pos, SEEK_SET);

// 获取文件总字节数
fseek(fp, 0L, SEEK_END);
long size = ftell(fp);

```

```c
void main ()
{
    FILE *fp = fopen("text.txt", "r+");
    char ch;

    ch = fgetc(fp);
    while (!feof(fp)) {
        if (ch >= 'A' && ch <= 'Z') {
            ch += 32;
            fseek(fp, -1L, SEEK_CUR);   // 指针回移，重置读写属性，切换为写

            fseek(fp, 0, SEEK_CUR);     // 切换回读
        }
        ch = fgetc(fp);
    }
    fclose(fp);
}

{
```

- 文件的索引：建立索引表查找
```c
// 索引文件

// 索引项：定长
typedef struct {
    int id;
    long offset;    // 在data.bin 中的字节偏移量
} IndexItem;

// 保存数据
void save_records(Student *student)
{
    FILE *f_data = fopen("data.bin", "wb");
    FILE *f_idx = fopen("index.bin", "wb");

    for (int i = 0; i < 3; i++) {
        // 获取当前数据文件写入位置偏移量
        long current_offset = ftell(f_data);
        // 写入主数据
        fwrite(&student[i], sizeof(Student), 1, f_data);
        // 构建索引项，写入索引文件
        IndexItem item = {student[i].id, current_offset};
        fwrite(&item, sizeof(IndexItem), 1, f_idx);
    }
    fclose(f_data);
    fclose(f_idx);
}

// 读取数据，通过索引随机访问
void find_student_by_id(int id)
{
    FILE *f_id = fopen("index.bin", "rb");
    FILE *f_data = fopen("data.bin", "rb");

    IndexItem item;
    int found = 0;
    
    while (fread(&item, sizeof(IndexItem), 1, f_id)) {
        if (item.id == id) {
            fseek(f_data, item.offset, SEEK_SET);

            Student s;
            fread(&s, sizeof(Student), 1, f_data);
            printf("查询成功！\n");
            break;
        }
    }

    if (!found)
        printf("未找到\n");

    fclose(f_data);
    fclose(f_id);
}
```

- 文件错误
    - feof(fp): 是否读到文件末尾
    - ferror(fp): 检查最近一次操作是否出现读写错误

```c
// 循环结束可能是EOF或者Error
```

- 错误诊断 
    - 使用全局变量 errno <errno.h>
        - 一个整数，记录最近一次系统调用的错误码：2（ENOENT，无此文件） 13（EACCES，无权限）
        - 使用strerror(errno)可以将错误码转换为文字描述
    - 使用perror() 输出错误信息 `void perror(const char *str)`
        - str为自定义错误信息前缀，通常描述错误来源，与errno中的错误信息一起输出
        - 输出：`str: errno`
        - 相当于 `fprintf(stderr, "%s:%s", str, strerror(errno)) ` 
```c
#include <errno.h>
#include <string.h>

FILE *fp = fopen(filename, "r");

if (fp == NULL) {
    perror("无法打开文件");

    fprintf(stderr, "错误码：%d 描述:%s\n", errno, strerror(errno));
}

```

- 结束检查函数 `int feof(FILE *stream)`
    - 读写指针越过文件尾，返回非零值
    - feof只有在读取失败的时候会变真，不是读取前，所以即使文件为空，也要先读再判断
